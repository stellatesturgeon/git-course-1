Команды для терминала

cd папка // переход в папку
mkdir папка // создание папки
pwd // показвает, где нахожусь в терминале
ls // показывает, что находится в папке
ls -la // отображение скрытых папок
echo "Some text" > file.txt // создает файл
cat название файла // чтение файла
rm название файла // удаление файла
clear // очищает терминал
------------------------------------------------
Команды для гита

git --version // версия гита
git config --global user.name "какое-то имя" // устанавливает имя автора
git config --global user.email mail // устанавливает почту автора
git config --list // получение всех настроек гита
git init // создание репозитория
git branch -m "имя ветки" // сменить наименование ветки
git status // показывает файлы, которые были изменены
git add название файла // добавление в индекс измененного файла
git add . // добавление в индекс всех измененных файлов
git commit -m "что сделано, что добавляем" // сохраняет текущую версию проекта (-m - используется для того, чтобы сразу указать сообщение)
git checkout <commit hash> // переход к определенной версии
git checkout <branch name> // переход в определенную версию проекта по названию ветки (переключает на ветку)
git log // просмотр истории изменений (коммитов)
git cat-file -t 61e1886 // посмотреть тип коммита (после -t идет первые 7 символов из коммита)
git cat-file -p 61e1886 // посмотреть содерживое коммита
